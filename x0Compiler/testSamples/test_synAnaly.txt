main
{
    int n;
    int cnt;
    int vis[100];
    int i;
    int j;

    read n;
    for (#i = 0; i <= n; #i = i + 1)
    {
        #vis[i] = 0;
    }
 
    #cnt = 0;
    for(#i = 2; i <= n; #i = i + 1)
    {
        if(vis[i] == 0)
        {
            write i;
        }
        for(#j = i; j <= n; #j = j + i)
        {
            #vis[j] = 1;
        }
    }
}
